Consider we have managers list and based on the selection of the manager from the dropdown list we need to load the employees working under him in the second list.

var app = angular.module('myApp', []);
angular.module('myApp').filter('myFilter', function() {
  return function(input, selectedManager) {
   
    input = input || '';
     if(selectedManager == null){
        return input;
    }else{
        var out = new Array();  
        selectedManager = JSON.parse(selectedManager);
        for (var i = 0; i < input.length; i++) {
            var item = input[i];
            if(item.managerId == selectedManager.id){
                out.push(item);
            }
        }
        return out;
    }
  };
});
