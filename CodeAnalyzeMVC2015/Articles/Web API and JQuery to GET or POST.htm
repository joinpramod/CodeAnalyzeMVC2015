We use Web API by calling the Web Api controller rather than project controller

Sample API controller is shown below

        public class EmployeesController : ApiController

        {

            EmployeeEntities db = new EmployeeEntities();

            public IEnumerable<emploee> Get()

            {

                return db.Employees.ToList();

            }

        } 

 

Sample model is as shown below

        public class Employee

        {

            public int ID { get; set; }

            public string Name { get; set; }

            public string Address { get; set; }

        }

        public class EmployeeEntities : DbContext

        {

            public DbSet<employee> Employees { get; set; }

        }

 

 

GET Method Jquery making request to web api is as shown below

        $(document).ready(function () {

        $.ajax({

            url: "http://localhost:49493/api/Values",

            type: "Get",

            success: function (data) {

                for (var i = 0; i < data.length; i++) {

                    $(data[i].Name + " " + data[i].Address).appendTo("#tbPerson");                  

                }

            },

            error: function (msg) { alert(msg); }

        });

    });

 

POST Method of Jquery

$(document).ready(function () {

    $("#submit").click(function () {

        var name = $("#Name").val();

        var address = $("#Address").val();

        var dob = $("#DOB").val();

        $.ajax({

            url: "http://localhost:49493/api/Values",

            type: "Post",

            data: JSON.stringify([name, address, dob]),

            contentType: 'application/json; charset=utf-8',

            success: function (data) { },

            error: function () { alert('error'); }

        });

    });

});
