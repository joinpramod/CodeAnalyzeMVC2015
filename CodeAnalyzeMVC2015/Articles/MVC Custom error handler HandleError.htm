As in classic asp.net application few of the exception handlers are similar as in defining custom error in web.config

 

Common methods similar to ASP.NET

Web.config

<customErrors defaultRedirect="Error.cshtml" mode="On">

 

 

Global.asax

        protected void Application_Error(object sender, EventArgs e)

        {

            Exception exception = Server.GetLastError();           

        }

 

 

 

 MVC Error handling

public class HomeController : Controller

        {

            protected override void OnException(ExceptionContext filterContext)

            {

                Exception ex = filterContext.Exception;

                filterContext.ExceptionHandled = true;

                //process exception

            }

        }

 

 

      OR

        public class HomeController : Controller

        {

            [HandleError()]

            public ActionResult SomeError()

            {

                throw new Exception("test");

            }

        }
