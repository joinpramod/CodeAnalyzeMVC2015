<script type="text/javascript">
    var mapPointers = [{ 'title': 'New York', 'lat': '40.730610', 'lng': '-73.935242', 'description': 'Google Map Testing for New York' }, ];;
    //the data for map pointers can be got from controller in a ViewBag and assigned here
    window.onload = function () {
        var googleMap = {
            center: new google.maps.LatLng(mapPointers[0].lat, mapPointers[0].lng),
            zoom: 10,
            mapTypeId: google.maps.MapTypeId.ROADMAP
        };
        var mappointerDetailsWindow = new google.maps.InfoWindow();
        var map = new google.maps.Map(document.getElementById("mapArea"), googleMap);
        for (i = 0; i < mapPointers.length; i++) {
            var data = mapPointers[i]
            var latLong = new google.maps.LatLng(data.lat, data.lng);
            var mapPointer = new google.maps.Marker({
                position: latLong,
                map: map,
                title: data.title
            });
            (function (mapPointer, data) {
                google.maps.event.addListener(mapPointer, "click", function (e) {
                    mappointerDetailsWindow.setContent(data.description);
                    mappointerDetailsWindow.open(map, mapPointer);
                });
            })(mapPointer, data);
        }
    }
</script>
