1. In This article we will see how to validate file upload control for file types and file size using jquery. Since this is a simple jquery we will directly look at the code without much explaning as code pretty straight forward. 

    <script type="text/javascript">
    $(document).ready(function () {
        $("#<%=form1.ClientID%>").validate({
            errorElement: "em",
            errorPlacement: function (error, element) {
                error.insertAfter(element.parent());
            },
            errorClass: "hasError"
        });
        //validationffor file type
        $("#<%=FileUpload1.ClientID%>").rules('add', {
            required: true,
            accept: "doc/docx/xls/xlsx",
            fileSize: true,
            messages: {
                accept: "Invalid file format"
            }
        });
        //validation for file size
        $.validator.addMethod("fileSize", function (value, element) {
            files = element.files;
            size = files[0].size;
            if (size > 61489) {
                return false;
            }
            return true;
        }, 'file should be upto 60 kb');
        $(document).on("change", function () {
            $("#<%=form1.ClientID%>").valid();
        });
    });

    </script>
