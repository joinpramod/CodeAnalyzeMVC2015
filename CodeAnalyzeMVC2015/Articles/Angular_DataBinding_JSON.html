1. In this article we will be binding JSON data to datatable plugin in AngularJS. It is considered that we have a controller returning JSON result containing a list of objects. We will be using jquery widgets. Take a look at the View code below and comments in between for more understanding 

    
    @section scripts {  
        <!--link all jquery widget css and javascripts as shown below-->
       <link href = "~/Content/jqx.base.css" rel="stylesheet" / >  
       <script src = "https://ajax.googleapis.com/ajax/libs/angularjs/1.4.7/angular.min.js"></script> 
       <script src = "~/Scripts/jqxcore.js"></script> 
       <script src = "~/Scripts/jqxangular.js"></script>
       <script src = "~/Scripts/jqxbuttons.js"></script> 
       <script src = "~/Scripts/jqxscrollbar.js"></script> 
       <script src = "~/Scripts/jqxdata.js"></script> 
       <script src = "~/Scripts/jqxdatatable.js"></script> 
       <script type = "text/javascript">
            var myApp = angular.module('myApp', ['jqwidgets']);
            myApp.controller('DataTableCtrl', ['$scope', function ($scope) {
                $scope.gridSettings = {
                    width: 1000,
                    pageable: true,
                    pagerButtonsCount: 10,
                    source: new $.jqx.dataAdapter({
                        dataType: "json",
                        dataFields: [{
                            name: 'EmpName',
                            type: 'string'
                        }, {
                            name: 'EmpEmail',
                            type: 'string'
                        }, {
                            name: 'EmpZipCode',
                            type: 'string'
                        }, {
                            name: 'EmpPhone',
                            type: 'string'
                        }, {
                            name: 'EmpManager',
                            type: 'string'
                        }],
                        id: 'EmpID',
                        url: 'GetEmployeeList'
                        <!--This is the Action in Controller-->
                    }),
                    columnsResize: true,
                    columns: [{
                        text: 'Name',
                        dataField: 'EmpName',
                        width: 200
                    }, {
                        text: 'Email',
                        dataField: 'EmpEmail',
                        width: 200
                    }, {
                        text: 'ZipCode',
                        dataField: 'EmpZipCode',
                        width: 200
                    }, {
                        text: 'Phone',
                        dataField: 'EmpPhone',
                        width: 200
                    }, {
                        text: 'Manager',
                        dataField: 'EmpManager',
                        width: 200
                    }]
                };
            }]);
       </script>  
    } 

  <div ng-app="myApp" ng-controller="DataTableCtrl">  
  <jqx-data-table jqx-settings="gridSettings"></jqx-data-table> 
  </div>
