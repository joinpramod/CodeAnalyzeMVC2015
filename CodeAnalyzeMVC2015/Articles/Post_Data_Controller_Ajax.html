1. To implement this we need to install jQuery unobtrusive ajax Nuget Package i.e. Microsoft.jquery.Unobtrusive.Ajax. After this we should find jquery.unobtrusive-ajax.js and jquery.unobtrusive-ajax.min.js in the scripts folder of the project. Considering we have required controller to accept the from request post and handle it, we will look at the view directly to implement this, please looks at the comments for more better understanding 

    @model AjaxBeginFormwithMVC.Models.EmpModel  
  
<script src="~/Scripts/jquery-1.10.2.js"></script>  
<script src="~/Scripts/jquery.unobtrusive-ajax.js"></script>  

<div id="divEmp">  
  
    //make sure we pass the div id here while beginning the form, we need to mention Ajax related configurations here.
    //With this it means the div portion only is posted to server and not entire page and hence avoiding whole refresh and updating to Ajax
    //EmpDetails is controller here

    @using (Ajax.BeginForm("EmpDetails", "Home", new AjaxOptions { HttpMethod = "POST", UpdateTargetId = "divEmp" }))  
    {  
        @Html.AntiForgeryToken()  
  
        <div>    
   
            @Html.ValidationSummary(true, "")  

            <div>  
                @Html.LabelFor(model => model.FirstName)  
                <div>  
                    @Html.EditorFor(model => model.FirstName)  
                    @Html.ValidationMessageFor(model => model.FirstName, "")  
                </div>  
            </div>  
  
            <div>  
                @Html.LabelFor(model => model.LastName)  
                <div>  
                    @Html.EditorFor(model => model.LastName)  
                    @Html.ValidationMessageFor(model => model.LastName, "")  
                </div>  
            </div>  
  
            <div>  
                @Html.LabelFor(model => model.Address)  
                <div>  
                    @Html.EditorFor(model => model.Address)  
                    @Html.ValidationMessageFor(model => model.Address, "")  
                </div>  
            </div>  
  
            <div>  
                    <input type="submit" value="Save"/>  
            </div>  
       
            <div>  
                <div>  
                   @ViewBag.Records  
                </div>  
            </div>  
  
        </div>  
    }  
</div>  
<script src="~/Scripts/jquery.validate.min.js"></script>  
<script src="~/Scripts/jquery.validate.unobtrusive.min.js"></script>
