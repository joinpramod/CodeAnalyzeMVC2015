Following are the methods

ActionAsPdf

        public ActionResult DownloadActionAsPDF()

        {

            var model = new GeneratePDFModel();

            return new Rotativa.ActionAsPdf("GeneratePDF", model) { FileName = "test.pdf" };

        }

 

        public ActionResult DownloadViewPDF()

        {

        var model = new GeneratePDFModel();

        return new Rotativa.ViewAsPdf("GeneratePDF", model) { FileName = "test.pdf" };

        }

 

        public ActionResult DownloadPartialViewPDF()

        {

            var model = new GeneratePDFModel();

            return new Rotativa.PartialViewAsPdf("_PartialViewTest", model) { FileName = "test.pdf" };

        }
