Jquery DataTables has an ibuilt drag and drop feature which can be used in MVC

 

Example Model

        public class Employee

        {

            public int ID { get; set; }

            public int EmpOrder { get; set; }

            public string FirstName { get; set; }

            public string Address { get; set; }

            public string Email { get; set; }

        }

 

 

 

View looks as below

<!DOCTYPE html>

<html>   

    <head>  

          

        <script src="/Scripts/jQuery-1.5.1.min.js" type="text/javascript"></script>

        <script src="/Scripts/jQuery-ui-1.8.11.min.js" type="text/javascript"></script>

        <script src="/Scripts/jQuery.dataTables.min.js" type="text/javascript"></script>

        <script src="/Scripts/jQuery.dataTables.rowReordering.js" type="text/javascript"></script>

 

        <script type="text/javascript">

        $(document).ready(function () {

            $('#EmpTable').dataTable();

        });

        </script>

 

<script type="text/javascript">

    $(document).ready(function () {

        $('#EmpTable').dataTable().rowReordering({ actionURL: "/Employee/ReOrder" });

    });

</script>

 

 

    </head>

    <body>

                <table id="EmpTable">

                    <thead>

                        <tr>

                            <th></th>

                            <th>Employee Name</th>

                            <th>Address</th>

                            <th>EMail</th>

                        </tr>

                    </thead>

                    <tbody>

                    @foreach(var company in Model) {

                        <tr id="@company.FirstName">

                            <tr>@company.Address</tr>

                            <tr>@company.EMail</tr>

                        </tr>

                    }

                    </tbody>

                </table>

    </body>

</html>
