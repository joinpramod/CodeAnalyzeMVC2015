First of all, you will need to have table to in SQL server with a column of varbinary(max) where we will storing the  images in binary.

The controllers for uploading and viewing the images are as given below

  [HttpGet]

  public ActionResult ViewImage(int? id)

     {

        string mime;       

        byte[] fileBytes = null

        using (DocumentEntities databaseContext = new DocumentEntities())

        {

          var databaseDocument = databaseContext.Documents.FirstOrDefault(doc => doc.DocumentId == id.Value);

          if (databaseDocument != nulll

          {

             fileBytes = databaseDocument.FileContent;

             mime = databaseDocument.Type;

          }

        }

        return File(fileBytes, mime);

     }

 

 

        [[HttpPost]

        public ActionResult Upload()

        {

           

            if (Request.Files.Count == 1)

            {

                var name = Request.Files[0].FileName;

                var size = Request.Files[0].ContentLength;

                var type = Request.Files[0].ContentType;

                Stream fileStream = Request.Files[0].InputStream;

 

                 byte[] documentBytes = new byte[fileStream.Length];

                 fileStream.Read(documentBytes, 0, documentBytes.Length);

 

                 Document databaseDocument = new Document

                 {

                     CreatedOn = DateTime.Now,

                     FileContent = documentBytes,

                     Name = name,

                     Size = size,

                     Type = type

                 };

 

                 using (DocumentEntities databaseContext = new DocumentEntities())

                 {

                     databaseContext.Documents.Add(databaseDocument);

                     databaseContext.SaveChanges();

                 }

            }

            return View();

        }
