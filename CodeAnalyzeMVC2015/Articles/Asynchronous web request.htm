With new Entity Framework 6 and .net 4.5 async and await keywards have been added to web api to make asynchronous requests from the MVC client application

 

We can implement this asynchronous stuff either directly in MVC Controller Or WEB API Controller

 

Suppose if we are implementing this in WEB API controller then below is the code 

public class MyWebAPIController : ApiController

    {

        DBEntities _db = new DBEntities(); 

 

        public async Task<IHttpActionResult> Create(EmployeeMaster emp)

        {

            _db.EmployeeMaster.Add(emp);

            await _db.SaveChangesAsync();

            return Created<EmployeeMaster>("api/Employee/" + emp.Id, emp);

        }

 

        public async Task<IHttpActionResult> Read()

        {

            var data = await _db.EmployeeMaster.ToListAsync();

            return data;

        }

 

        public async Task<IHttpActionResult> Update(EmployeeMaster emp, int Id)

        {

                var record = await _db.EmployeeMaster.Where(f => f.id == Id).FirstOrDefaultAsync();

                record.firstname = emp.firstname;

                record.lastname = emp.lastname;

                await _db.SaveChangesAsync();

                return;

        }

 

        public async Task<IHttpActionResult> Delete(Int32 Id)

        {

            var record = await _db.EmployeeMaster.Where(f => f.id == Id).FirstOrDefaultAsync();

            _db.EmployeeMaster.Remove(record);

                 await _db.SaveChangesAsync();         

        }

    }

 

 

Suppose if we are implementing this directly in Controller of MVC application then

         public class MyMVCAsyncController : Controller

        {

            DBEntities _db = new DBEntities();

            public async Task<actionresult> GetData()

            {

                return View("View", await _db.EmployeeMaster.ToListAsync());

            }

        }

 

The above can also be implemented by inheriting from AsyncController controller class
