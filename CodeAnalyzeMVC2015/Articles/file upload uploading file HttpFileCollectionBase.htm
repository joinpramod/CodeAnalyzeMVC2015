Controller

 

        [HttpPost]

        public ActionResult UploadFiles()

        {

            string path = @"C:\Files\";

            HttpPostedFileBase files = Request.Files["files"];

            if (files != null)

                files.SaveAs(path + files.FileName);

            return RedirectToAction("Index");

        }

 

 

Or file can be made available as parameter

        [HttpPost]

        public ActionResult UploadFiles(HttpPostedFileBase files)

        {

            string path = @"C:\Files\";

            if (files != null)

                files.SaveAs(path + files.FileName);

            return RedirectToAction("Index");

        }
