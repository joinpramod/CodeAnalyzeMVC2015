1. In this article we will create a dropdownlist with multiple select options using checkbox for each item in dropdownlist in asp.net mvc.
This can be also similar to ListBox but we will have checkbox for each item. We will bind the data to the dropdownlist from the controller using the model.
We will be using jquery and bootstrap to implement this feature. So we need to refer the scripts and css files. 
With the new scripts users can see the selected items.
  2. View Demo of Code

  3. Controller 
    
public ActionResult Index()
{
    List<SelectListItem> items = GetItems();

    return View(items);
}

public ActionResult PostValues(FormCollection form)
{
    string lbComp = form["lbComp"];
    ViewBag.Message += lbComp;
    List<SelectListItem> items = GetItems();
    return View(items);
}

private static List<SelectListItem> GetItems()
{
    List<SelectListItem> items = new List<SelectListItem>();

    items.Add(new SelectListItem
    {
        Text = "Microsoft",
        Value = "1"
    });

    items.Add(new SelectListItem
    {
        Text = "Apple",
        Value = "2"
    });

    items.Add(new SelectListItem
    {
        Text = "IBM",
        Value = "3"
    });

    items.Add(new SelectListItem
    {
        Text = "Oracle",
        Value = "4"
    });

    items.Add(new SelectListItem
    {
        Text = "Google",
        Value = "5"
    });
    return items;
}
